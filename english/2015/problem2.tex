\documentclass[varwidth]{standalone}

\usepackage{amsmath}

\begin{document}
    A convex hexagon $ABCDEF$ is such that
    \begin{align*}
        AB & = BC & CD & = DE & EF & = FA
    \end{align*}
    and
    \begin{align*}
        \angle ABC & = 2 \angle AEC & \angle CDE & = 2 \angle CAE & \angle EFA & = 2 \angle ACE.
    \end{align*}
    Show that $AD$, $CF$, and $EB$ are concurrent.
\end{document}
